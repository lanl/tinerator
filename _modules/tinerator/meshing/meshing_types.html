# Standard primitives in ExodusII format
class ExodusElements(Enum):
    # ExodusII API: https://gsjaardema.github.io/seacas-docs/exodusII-new.pdf
    # pp. 16
    LINE = "Bar2"
    QUAD = "Quad4"
    TRIANGLE = "Tri3"
    TETRA = "Tet4"  # Tetrahedron
    WEDGE = "Wedge6"  # Prism
    HEX = "Hex8"


_ex = ExodusElements

# Mapping between (sorted low-high) face
